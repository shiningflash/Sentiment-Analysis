```python
def removeAndTag(sen):
    sen = sen.replace('&quot;3', '')
    sen = sen.replace('&quot;', '')
    sen = sen.replace('&lt;3', '')
    sen = sen.replace('&lt;', '')
    sen = sen.replace('&gt;', '')
    return sen
```

```python
def removeMentioned(sen):
```

```python
def convertShortcutOfNot(sen):
```

```python
def removeMultipleSpaceDashAndDot(sen):
    sen = re.sub(r'[.]+', '.', sen)
    sen = re.sub(r'\s+', ' ', sen)
    sen = re.sub(r'[-]+', ' ', sen)
    return sen
```
